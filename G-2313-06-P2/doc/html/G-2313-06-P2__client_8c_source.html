502 long IRCInterface_Connect(char *nick, char *user, char *realname, char *password, char *server, int port, boolean ssl)
503 {
504  struct sockaddr_in socket_address;
505  struct hostent *server_info;
506  char *command, *msgNick, *msgUser;
507  syslog (LOG_INFO, "[CLIENTE] Inicia conexion");
508  if(!nick||!user||!realname||!server){
509  return IRCERR_NOCONNECT;
510  } else {
511  /* Se crea el socket que vamos a usar */
512  if((socket_desc = socket(AF_INET, SOCK_STREAM, 0)) < 0 ){
513  syslog(LOG_ERR, "[CLIENTE] ERROR creando socket");
514  return IRCERR_NOCONNECT;
515  } else {
516  syslog(LOG_ERR, "[CLIENTE] Socket creado OK");
517  }
518 
519  server_info = gethostbyname(server);
520  if(server_info == NULL) {
521  syslog(LOG_ERR, "[CLIENTE] ERROR obteniendo hostname");
522  return IRCERR_NOCONNECT;
523  } else {
524  syslog(LOG_ERR, "[CLIENTE] Hostname obtenido OK");
525  }
526 
527  bzero((char *) &socket_address, sizeof (socket_address));
528  socket_address.sin_family = AF_INET;
529  bcopy((char *) server_info->h_addr, (char *) &socket_address.sin_addr.s_addr, server_info->h_length);
530  socket_address.sin_port = htons(port);
531 
532  if(connect(socket_desc, (struct sockaddr*) &socket_address, sizeof (socket_address)) < 0) {
533  syslog(LOG_ERR, "[CLIENTE] ERROR conectando con el servidor");
534  return IRCERR_NOCONNECT;
535  } else {
536  syslog(LOG_ERR, "[CLIENTE] Conexion con servidor OK");
537  }
538  pthread_create(&thread_response, NULL, client_function_response, NULL);
539 
540  /* Ya conectados, parseamos nick/user y lo enviamos */
541  IRCMsg_Nick(&msgNick, NULL, nick, NULL);
542  IRCMsg_User(&msgUser, NULL, user, server, realname);
543  /*IRCMsg_Pass(&msgPass, NULL, password);*/
544  syslog(LOG_ERR, "[CLIENTE] Password: %s", password);
545  IRC_PipelineCommands(&command, msgNick, msgUser, NULL);
546  syslog(LOG_ERR, "[CLIENTE] Registro pipelined: %s", command);
547 
548  if(send(socket_desc, command, strlen(command), 0) < 0) {
549  return IRCERR_NOCONNECT;
550  }
551  IRCInterface_PlaneRegisterOutMessage(command);
552 
553  if(!nick_cliente){
554  syslog(LOG_INFO, "[CLIENTE] Nuevo nick almacenado");
555  nick_cliente = malloc(sizeof(nick));
556  strcpy(nick_cliente, nick);
557  }
558 
559  syslog(LOG_ERR, "[CLIENTE] Socket desc: %d", socket_desc);
560  free(command);
561  }
562 
563  pthread_create(&thread_ping, NULL, client_function_ping, NULL);
564 
565  return IRC_OK;
566 }
